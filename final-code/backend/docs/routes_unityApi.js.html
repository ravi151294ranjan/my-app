var express = require('express');
var router = express.Router();

const unityModel = require("../models/unityModel")


/**
 * Sends a HTTP GET request to get all asset.
 * </br>
 * @function getAllAssetForUnity
 * 
 * @path {get} path /asset/getAllAssetForUnity
 * @param req {Object} The req object represents the HTTP request.
 * @param res {Object} The res object represents the HTTP response.
 * @param {Function}  unitymodel.getAllAssetForUnity() This function is used to pass data from unity
 * @return {Object} Its retrun success or failure message with all data.
 */
router.post('/getAllAssetForUnity', (req, res) => {
    try {
        let data=req.body
        const vGetAllAssetDetails = unityModel.getAllAssetForUnity('assestdetails',data)
        vGetAllAssetDetails.then((data) => {
            if (data.length > 0) {
                let result = {
                    success: true,
                    data: data
                }
                res.status(200).send(result);
            } else {
                let result = {
                    success: false,
                    data: ''
                }
                res.status(200).send(result);
            }
        }).catch(err => {
            util.writeLog(`${err} -> get asset vGetAllAssetDetails Error`, 'get:/unity/getAllAssetForUnity');
            var error = new Error();
            error.success = false;
            error.status = 404;
            error.message = 'Asset not found';
            res.send(error);
        })
    } catch (err) {
        util.writeLog(`${err} -> get asset vGetAllAssetDetails Error`, 'get:/unity/getAllAssetForUnity');
        var error = new Error();
        error.success = false;
        error.status = 404;
        error.message = 'An internal error occurred. Please try again later';
        res.send(error);
    }
})
